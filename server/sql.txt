-- 1. 사용자 계정 생성
CREATE USER 'workmate'@'localhost' IDENTIFIED BY '12345';

-- 2. workmates 데이터베이스에 대한 모든 권한 부여
GRANT ALL PRIVILEGES ON workmates.* TO 'workmate'@'localhost';

-- 3. 권한 적용
FLUSH PRIVILEGES;


-- 4. 데이터 베이스 생성 workmates
create database workmates;

--5 임시 데이터베이스 user, email. 
CREATE TABLE USER (
  id INT AUTO_INCREMENT PRIMARY KEY,
  nickname VARCHAR(50) NOT NULL,
  password VARCHAR(255) NOT NULL,
  createdAt DATETIME DEFAULT CURRENT_TIMESTAMP,
  updatedAt DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE EMAIL (
  id INT AUTO_INCREMENT PRIMARY KEY,
  email VARCHAR(100) NOT NULL UNIQUE,
  user_id INT NOT NULL,
  createdAt DATETIME DEFAULT CURRENT_TIMESTAMP,
  updatedAt DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES USER(id) ON DELETE CASCADE
);

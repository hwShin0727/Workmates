-- 1. 사용자 계정 생성
CREATE USER 'workmate'@'localhost' IDENTIFIED BY '12345';

-- 2. workmates 데이터베이스에 대한 모든 권한 부여
GRANT ALL PRIVILEGES ON workmates.* TO 'workmate'@'localhost';

-- 3. 권한 적용
FLUSH PRIVILEGES;


-- 4. 데이터 베이스 생성 workmates
create database workmates;

--5 임시 데이터베이스 user, email. 
CREATE TABLE USER (
  user_id VARCHAR(255) NOT NULL PRIMARY KEY,       -- 사용자 아이디 (PK)
  user_code INT NOT NULL UNIQUE,                   -- 사용자 코드 (고유값)
  password VARCHAR(255) NOT NULL,                  -- 비밀번호
  nickname VARCHAR(255) NOT NULL,                  -- 닉네임
  group_name VARCHAR(255) NOT NULL,                -- 소속 (예약어 'group' 대신 'group_name' 사용)
  position VARCHAR(255),                           -- 직책 (NULL 허용)
  profile_description TEXT,                        -- 자기 소개 (멀티라인)
  user_setting TEXT NOT NULL                       -- 유저 개인 설정 (멀티라인)
);

CREATE TABLE EMAIL (
  user_id VARCHAR(255) NOT NULL,                   -- USER와의 관계 (FK)
  email VARCHAR(255) NOT NULL UNIQUE,              -- 사용자 이메일
  auth_token VARCHAR(255),                         -- 인증 토큰 (NULL 가능)

  PRIMARY KEY (user_id),                           -- user_id 기준으로 1:1 관계
  FOREIGN KEY (user_id) REFERENCES USER(user_id)   -- FK 제약
);
